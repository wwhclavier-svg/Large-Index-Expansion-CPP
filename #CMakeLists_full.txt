option(BUILD_FF_TESTS "Build finite field tests (requires FireFly)" ON)

if(BUILD_FF_TESTS)
    # 查找 FireFly 头文件
    find_path(FIREFLY_INCLUDE_DIR
        NAMES firefly/FFInt.hpp
        PATHS /usr/local/include /usr/include
        PATH_SUFFIXES firefly
    )
    # 查找 FireFly 库
    find_library(FIREFLY_LIBRARY
        NAMES firefly
        PATHS /usr/local/lib /usr/lib
    )
    # 查找 GMP 库（FireFly 依赖）
    find_library(GMP_LIBRARY NAMES gmp gmpxx)

    if(FIREFLY_INCLUDE_DIR AND FIREFLY_LIBRARY AND GMP_LIBRARY)
        message(STATUS "Found FireFly: ${FIREFLY_INCLUDE_DIR} / ${FIREFLY_LIBRARY}")
        set(HAVE_FIREFLY TRUE)
    else()
        message(WARNING "FireFly or GMP not found, disabling finite field tests")
        set(HAVE_FIREFLY FALSE)
    endif()
endif()

if(HAVE_FIREFLY)
    # 添加 FireFly 头文件路径
    include_directories(${FIREFLY_INCLUDE_DIR})

    # 生成有限域测试可执行文件
    add_executable(test_expandFF
        tests/test_expandFF.cpp
        ${COMMON_SOURCES}
    )
    target_link_libraries(test_expandFF PRIVATE
        ${FIREFLY_LIBRARY}
        ${GMP_LIBRARY}
        Eigen3::Eigen
    )
endif()
